#include <tunables/global>

# AppArmor profile for the EnOcean-MQTT-HA add-on (profile name MUST equal the add-on slug).
# The Home Assistant Supervisor loads this file (found next to config.yaml) and runs the container
# confined to it. It is the vanilla HA add-on template (s6-overlay v3 + bashio) plus only what this
# add-on intrinsically needs: MQTT/ser2net network, the EnOcean serial device, DNS, and the health
# check. The app installs into the base-python image's /usr/local Python, so there is no /app carve-out.
profile enocean_mqtt_ha flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>
  # DNS: resolve broker / ser2net / supervisor hostnames.
  #include <abstractions/nameservice>

  file,
  signal,
  # open the serial device node (/dev/tty*)
  capability dac_override,
  # MQTT + ser2net TCP + Supervisor API
  network,
  # HEALTHCHECK `pgrep -f enocean2mqtt` reads /proc of the daemon (same profile)
  ptrace (read) peer=enocean_mqtt_ha,

  # S6-Overlay v3 + bashio (globs; base:3.20 ships s6-overlay 3.2.3.0).
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  # in case uv --system placed the entry script under /usr/local/bin
  /usr/local/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/s6-overlay/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,
  /usr/lib/bashio/** ix,

  # Python site-packages: .so C-extensions are mmap'd, which needs 'm' (bare file, is not enough).
  # base-python installs Python (and our wheel) under /usr/local; /usr/lib kept for any apk python bits.
  /usr/local/lib/python3*/** mr,
  /usr/lib/python3*/** mr,

  # Add-on data (/data: generated .conf + sqlite DB & WAL) and the /config map (log, devices.yaml,
  # optional mapping files).
  /data/** rw,
  /config/** rw,

  # EnOcean USB transceiver (uart). A remote ser2net transceiver uses the network instead.
  /dev/ttyUSB* rw,
  /dev/ttyACM* rw,
  /dev/serial/** rw,
}
